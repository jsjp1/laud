cmake_minimum_required(VERSION 3.10)

project(laud VERSION 1.0 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

file(GLOB LIB_SOURCES laud/*.cpp)
add_library(laud_lib ${LIB_SOURCES})

target_include_directories(laud_lib PUBLIC ${CMAKE_SOURCE_DIR}/include)

find_library(PORTAUDIO_LIB portaudio REQUIRED)

add_executable(laud laud/main.cpp)

target_link_libraries(laud PRIVATE ${PORTAUDIO_LIB})
